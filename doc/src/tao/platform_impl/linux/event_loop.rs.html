51#[derive(Clone)]
52pub struct EventLoopWindowTarget<T> {
53  /// Gdk display
54  pub(crate) display: gdk::Display,
55  /// Gtk application
56  pub(crate) app: gtk::Application,
57  /// Window Ids of the application
58  pub(crate) windows: Rc<RefCell<HashSet<WindowId>>>,
59  /// Window requests sender
60  pub(crate) window_requests_tx: glib::Sender<(WindowId, WindowRequest)>,
61  /// Draw event sender
62  pub(crate) draw_tx: crossbeam_channel::Sender<WindowId>,
63  _marker: std::marker::PhantomData<T>,
64}
65
66impl<T> EventLoopWindowTarget<T> {
67  #[inline]
68  pub fn monitor_from_point(&self, x: f64, y: f64) -> Option<MonitorHandle> {
69    monitor::from_point(&self.display, x, y)
70  }
71  #[inline]
72  pub fn available_monitors(&self) -> VecDeque<MonitorHandle> {
73    let mut handles = VecDeque::new();
74    let display = &self.display;
75    let numbers = display.n_monitors();
76
77    for i in 0..numbers {
78      let monitor = MonitorHandle::new(display, i);
79      handles.push_back(monitor);
80    }
81
82    handles
83  }
84
85  #[inline]
86  pub fn primary_monitor(&self) -> Option<RootMonitorHandle> {
87    let monitor = self.display.primary_monitor();
88    monitor.and_then(|monitor| {
89      let handle = MonitorHandle { monitor };
90      Some(RootMonitorHandle { inner: handle })
91    })
92  }
93
94  #[cfg(feature = "rwh_05")]
95  pub fn raw_display_handle_rwh_05(&self) -> rwh_05::RawDisplayHandle {
96    if self.is_wayland() {
97      let mut display_handle = rwh_05::WaylandDisplayHandle::empty();
98      display_handle.display = unsafe {
99        gdk_wayland_sys::gdk_wayland_display_get_wl_display(self.display.as_ptr() as *mut _)
100      };
101      rwh_05::RawDisplayHandle::Wayland(display_handle)
102    } else {
103      let mut display_handle = rwh_05::XlibDisplayHandle::empty();
104      unsafe {
105        if let Ok(xlib) = x11_dl::xlib::Xlib::open() {
106          let display = (xlib.XOpenDisplay)(std::ptr::null());
107          display_handle.display = display as _;
108          display_handle.screen = (xlib.XDefaultScreen)(display) as _;
109        }
110      }
111
112      rwh_05::RawDisplayHandle::Xlib(display_handle)
113    }
114  }
115
116  #[cfg(feature = "rwh_06")]
117  pub fn raw_display_handle_rwh_06(&self) -> Result<rwh_06::RawDisplayHandle, rwh_06::HandleError> {
118    if self.is_wayland() {
119      let display = unsafe {
120        gdk_wayland_sys::gdk_wayland_display_get_wl_display(self.display.as_ptr() as *mut _)
121      };
122      let display = unsafe { std::ptr::NonNull::new_unchecked(display) };
123      let display_handle = rwh_06::WaylandDisplayHandle::new(display);
124      Ok(rwh_06::RawDisplayHandle::Wayland(display_handle))
125    } else {
126      unsafe {
127        if let Ok(xlib) = x11_dl::xlib::Xlib::open() {
128          let display = (xlib.XOpenDisplay)(std::ptr::null());
129          let screen = (xlib.XDefaultScreen)(display) as _;
130          let display = std::ptr::NonNull::new_unchecked(display as _);
131          let display_handle = rwh_06::XlibDisplayHandle::new(Some(display), screen);
132          Ok(rwh_06::RawDisplayHandle::Xlib(display_handle))
133        } else {
134          Err(rwh_06::HandleError::Unavailable)
135        }
136      }
137    }
138  }
139
140  pub fn is_wayland(&self) -> bool {
141    self.display.backend().is_wayland()
142  }
143
144  pub fn is_x11(&self) -> bool {
145    self.display.backend().is_x11()
146  }
147
148  #[inline]
149  pub fn cursor_position(&self) -> Result<PhysicalPosition<f64>, ExternalError> {
150    util::cursor_position(self.is_wayland())
151  }
152
153  #[inline]
154  pub fn set_progress_bar(&self, progress: ProgressBarState) {
155    if let Err(e) = self
156      .window_requests_tx
157      .send((WindowId::dummy(), WindowRequest::ProgressBarState(progress)))
158    {
159      log::warn!("Fail to send update progress bar request: {e}");
160    }
161  }
162
163  #[inline]
164  pub fn set_badge_count(&self, count: Option<i64>, desktop_filename: Option<String>) {
165    if let Err(e) = self.window_requests_tx.send((
166      WindowId::dummy(),
167      WindowRequest::BadgeCount(count, desktop_filename),
168    )) {
169      log::warn!("Fail to send update progress bar request: {e}");
170    }
171  }
172
173  #[inline]
174  pub fn set_theme(&self, theme: Option<Theme>) {
175    if let Err(e) = self
176      .window_requests_tx
177      .send((WindowId::dummy(), WindowRequest::SetTheme(theme)))
178    {
179      log::warn!("Fail to send update theme request: {e}");
180    }
181  }
182}

184pub struct EventLoop<T: 'static> {
185  /// Window target.
186  window_target: RootELW<T>,
187  /// User event sender for EventLoopProxy
188  pub(crate) user_event_tx: crossbeam_channel::Sender<Event<'static, T>>,
189  /// Event queue of EventLoop
190  events: crossbeam_channel::Receiver<Event<'static, T>>,
191  /// Draw queue of EventLoop
192  draws: crossbeam_channel::Receiver<WindowId>,
193  /// Boolean to control device event thread
194  run_device_thread: Option<Rc<AtomicBool>>,
195}
196
197#[derive(Default, Debug, Clone, PartialEq, Eq, Hash)]
198pub(crate) struct PlatformSpecificEventLoopAttributes {
199  pub(crate) any_thread: bool,
200  pub(crate) app_id: Option<String>,
201}
202
203impl<T: 'static> EventLoop<T> {
204  pub(crate) fn new(attrs: &PlatformSpecificEventLoopAttributes) -> EventLoop<T> {
205    if !attrs.any_thread {
206      assert_is_main_thread("new_any_thread");
207    }
208
209    let context = MainContext::default();
210    context
211      .with_thread_default(|| {
212        EventLoop::new_gtk(attrs.app_id.as_deref()).expect("Failed to initialize gtk backend!")
213      })
214      .expect("Failed to initialize gtk backend!")
215  }
216
217  fn new_gtk(app_id: Option<&str>) -> Result<EventLoop<T>, Box<dyn Error>> {
218    let context = MainContext::default();
219    let app = gtk::Application::new(app_id, gio::ApplicationFlags::empty());
220    let app_ = app.clone();
221    let cancellable: Option<&Cancellable> = None;
222    app.register(cancellable)?;
223
224    // Send StartCause::Init event
225    let (event_tx, event_rx) = crossbeam_channel::unbounded();
226    let (draw_tx, draw_rx) = crossbeam_channel::unbounded();
227    let event_tx_ = event_tx.clone();
228    app.connect_activate(move |_| {
229      if let Err(e) = event_tx_.send(Event::NewEvents(StartCause::Init)) {
230        log::warn!("Failed to send init event to event channel: {}", e);
231      }
232    });
233    let draw_tx_ = draw_tx.clone();
234    let user_event_tx = event_tx.clone();
235
236    // Create event loop window target.
237    let (window_requests_tx, window_requests_rx) = glib::MainContext::channel(Priority::default());
238    let display = gdk::Display::default()
239      .expect("GdkDisplay not found. This usually means `gkt_init` hasn't called yet.");
240    let window_target = EventLoopWindowTarget {
241      display,
242      app,
243      windows: Rc::new(RefCell::new(HashSet::new())),
244      window_requests_tx,
245      draw_tx: draw_tx_,
246      _marker: std::marker::PhantomData,
247    };
248
249    // Spawn x11 thread to receive Device events.
250    let run_device_thread = if window_target.is_x11() {
251      let (device_tx, device_rx) = glib::MainContext::channel(glib::Priority::default());
252      let user_event_tx = user_event_tx.clone();
253      let run_device_thread = Rc::new(AtomicBool::new(true));
254      let run = run_device_thread.clone();
255      device::spawn(device_tx);
256      device_rx.attach(Some(&context), move |event| {
257        if let Err(e) = user_event_tx.send(Event::DeviceEvent {
258          device_id: DEVICE_ID,
259          event,
260        }) {
261          log::warn!("Fail to send device event to event channel: {}", e);
262        }
263        if run.load(Ordering::Relaxed) {
264          glib::ControlFlow::Continue
265        } else {
266          glib::ControlFlow::Break
267        }
268      });
269      Some(run_device_thread)
270    } else {
271      None
272    };
273
274    let mut taskbar = TaskbarIndicator::new();
275    let is_wayland = window_target.is_wayland();
276
277    // Window Request
278    window_requests_rx.attach(Some(&context), move |(id, request)| {
279      if let Some(window) = app_.window_by_id(id.0) {
280        match request {
281          WindowRequest::Title(title) => window.set_title(&title),
282          WindowRequest::Position((x, y)) => window.move_(x, y),
283          WindowRequest::Size((w, h)) => window.resize(w, h),
284          WindowRequest::SizeConstraints(constraints) => {
285            util::set_size_constraints(&window, constraints);
286          }
287          WindowRequest::Visible(visible) => {
288            if visible {
289              window.show_all();
290            } else {
291              window.hide();
292            }
293          }
294          WindowRequest::Focus => {
295            window.present_with_time(gdk::ffi::GDK_CURRENT_TIME as _);
296          }
297          WindowRequest::Resizable(resizable) => window.set_resizable(resizable),
298          WindowRequest::Closable(closable) => window.set_deletable(closable),
299          WindowRequest::Minimized(minimized) => {
300            if minimized {
301              window.iconify();
302            } else {
303              window.deiconify();
304            }
305          }
306          WindowRequest::Maximized(maximized, resizable) => {
307            if maximized {
308              let maximize_process = util::WindowMaximizeProcess::new(window.clone(), resizable);
309              glib::idle_add_local_full(glib::Priority::DEFAULT_IDLE, move || {
310                let mut maximize_process = maximize_process.borrow_mut();
311                maximize_process.next_step()
312              });
313            } else {
314              window.unmaximize();
315            }
316          }
317          WindowRequest::DragWindow => {
318            if let Some(cursor) = window
319              .display()
320              .default_seat()
321              .and_then(|seat| seat.pointer())
322            {
323              let (_, x, y) = cursor.position();
324              window.begin_move_drag(1, x, y, 0);
325            }
326          }
327          WindowRequest::DragResizeWindow(direction) => {
328            if let Some(cursor) = window
329              .display()
330              .default_seat()
331              .and_then(|seat| seat.pointer())
332            {
333              let (_, x, y) = cursor.position();
334              window.begin_resize_drag(
335                direction.to_gtk_edge(),
336                1,
337                x,
338                y,
339                gtk::gdk::ffi::GDK_CURRENT_TIME as _,
340              );
341            }
342          }
343          WindowRequest::Fullscreen(fullscreen) => match fullscreen {
344            Some(f) => {
345              if let Fullscreen::Borderless(m) = f {
346                if let Some(monitor) = m {
347                  let display = window.display();
348                  let monitor = monitor.inner;
349                  let monitors = display.n_monitors();
350                  for i in 0..monitors {
351                    let m = display.monitor(i).unwrap();
352                    if m == monitor.monitor {
353                      let screen = display.default_screen();
354                      window.fullscreen_on_monitor(&screen, i);
355                    }
356                  }
357                } else {
358                  window.fullscreen();
359                }
360              }
361            }
362            None => window.unfullscreen(),
363          },
364          WindowRequest::Decorations(decorations) => window.set_decorated(decorations),
365          WindowRequest::AlwaysOnBottom(always_on_bottom) => {
366            window.set_keep_below(always_on_bottom)
367          }
368          WindowRequest::AlwaysOnTop(always_on_top) => window.set_keep_above(always_on_top),
369          WindowRequest::WindowIcon(window_icon) => {
370            if let Some(icon) = window_icon {
371              window.set_icon(Some(&icon.inner.into()));
372            }
373          }
374          WindowRequest::UserAttention(request_type) => {
375            window.set_urgency_hint(request_type.is_some())
376          }
377          WindowRequest::SetSkipTaskbar(skip) => {
378            window.set_skip_taskbar_hint(skip);
379            window.set_skip_pager_hint(skip)
380          }
381          WindowRequest::BackgroundColor(css_provider, color) => {
382            unsafe { window.set_data("background_color", color) };
383
384            let style_context = window.style_context();
385            style_context.remove_provider(&css_provider);
386
387            if let Some(color) = color {
388              let theme = format!(
389                r#"
390                  window {{
391                    background-color:  rgba({},{},{},{});
392                    }}
393                    "#,
394                color.0,
395                color.1,
396                color.2,
397                color.3 as f64 / 255.0
398              );
399              let _ = css_provider.load_from_data(theme.as_bytes());
400              style_context.add_provider(&css_provider, gtk::STYLE_PROVIDER_PRIORITY_APPLICATION);
401            };
402          }
403          WindowRequest::SetVisibleOnAllWorkspaces(visible) => {
404            if visible {
405              window.stick();
406            } else {
407              window.unstick();
408            }
409          }
410          WindowRequest::CursorIcon(cursor) => {
411            if let Some(gdk_window) = window.window() {
412              let display = window.display();
413              match cursor {
414                Some(cr) => {
415                  gdk_window.set_cursor(Cursor::from_name(&display, cr.to_str()).as_ref())
416                }
417                None => gdk_window
418                  .set_cursor(Cursor::for_display(&display, CursorType::BlankCursor).as_ref()),
419              }
420            };
421          }
422          WindowRequest::CursorPosition((x, y)) => {
423            if let Some(cursor) = window
424              .display()
425              .default_seat()
426              .and_then(|seat| seat.pointer())
427            {
428              if let Some(screen) = GtkWindowExt::screen(&window) {
429                cursor.warp(&screen, x, y);
430              }
431            }
432          }
433          WindowRequest::CursorIgnoreEvents(ignore) => {
434            if ignore {
435              let empty_region = Region::create_rectangle(&RectangleInt::new(0, 0, 1, 1));
436              window
437                .window()
438                .unwrap()
439                .input_shape_combine_region(&empty_region, 0, 0);
440            } else {
441              window.input_shape_combine_region(None)
442            };
443          }
444          WindowRequest::ProgressBarState(_) => unreachable!(),
445          WindowRequest::BadgeCount(_, _) => unreachable!(),
446          WindowRequest::SetTheme(_) => unreachable!(),
447          WindowRequest::WireUpEvents {
448            transparent,
449            fullscreen,
450            cursor_moved,
451          } => {
452            window.add_events(
453              EventMask::POINTER_MOTION_MASK
454                | EventMask::BUTTON1_MOTION_MASK
455                | EventMask::BUTTON_PRESS_MASK
456                | EventMask::TOUCH_MASK
457                | EventMask::STRUCTURE_MASK
458                | EventMask::FOCUS_CHANGE_MASK
459                | EventMask::SCROLL_MASK,
460            );
461
462            let fullscreen = Rc::new(AtomicBool::new(fullscreen));
463            let fullscreen_ = fullscreen.clone();
464            window.connect_window_state_event(move |_window, event| {
465              let state = event.changed_mask();
466              if state.contains(WindowState::FULLSCREEN) {
467                fullscreen_.store(
468                  event.new_window_state().contains(WindowState::FULLSCREEN),
469                  Ordering::Relaxed,
470                );
471              }
472              glib::Propagation::Proceed
473            });
474
475            // Allow resizing unmaximized non-fullscreen undecorated window
476            let fullscreen_ = fullscreen.clone();
477            window.connect_motion_notify_event(move |window, event| {
478              if !window.is_decorated() && window.is_resizable() && !window.is_maximized() {
479                if let Some(window) = window.window() {
480                  let (cx, cy) = event.root();
481                  let (left, top) = window.position();
482                  let (w, h) = (window.width(), window.height());
483                  let (right, bottom) = (left + w, top + h);
484                  let border = window.scale_factor() * 5;
485                  let edge = crate::window::hit_test(
486                    (left, top, right, bottom),
487                    cx as _,
488                    cy as _,
489                    border,
490                    border,
491                  );
492
493                  let edge = match &edge {
494                    Some(e) if !fullscreen_.load(Ordering::Relaxed) => e.to_cursor_str(),
495                    _ => "default",
496                  };
497                  window.set_cursor(Cursor::from_name(&window.display(), edge).as_ref());
498                }
499              }
500              glib::Propagation::Proceed
501            });
502            window.connect_button_press_event(move |window, event| {
503              const LMB: u32 = 1;
504              if (is_wayland || !window.is_decorated())
505                && window.is_resizable()
506                && !window.is_maximized()
507                && event.button() == LMB
508              {
509                let (cx, cy) = event.root();
510                let (left, top) = window.position();
511                let (w, h) = window.size();
512                let (right, bottom) = (left + w, top + h);
513                let border = window.scale_factor() * 5;
514                let edge = crate::window::hit_test(
515                  (left, top, right, bottom),
516                  cx as _,
517                  cy as _,
518                  border,
519                  border,
520                )
521                .map(|d| d.to_gtk_edge())
522                // we return `WindowEdge::__Unknown` to be ignored later.
523                // we must return 8 or bigger, otherwise it will be the same as one of the other 7 variants of `WindowEdge` enum.
524                .unwrap_or(WindowEdge::__Unknown(8));
525                // Ignore the `__Unknown` variant so the window receives the click correctly if it is not on the edges.
526                match edge {
527                  WindowEdge::__Unknown(_) => (),
528                  _ => {
529                    // FIXME: calling `window.begin_resize_drag` uses the default cursor, it should show a resizing cursor instead
530                    window.begin_resize_drag(edge, LMB as i32, cx as i32, cy as i32, event.time())
531                  }
532                }
533              }
534
535              glib::Propagation::Proceed
536            });
537            window.connect_touch_event(move |window, event| {
538              if !window.is_decorated() && window.is_resizable() && !window.is_maximized() {
539                if let Some(window) = window.window() {
540                  if let Some((cx, cy)) = event.root_coords() {
541                    if let Some(device) = event.device() {
542                      let (left, top) = window.position();
543                      let (w, h) = (window.width(), window.height());
544                      let (right, bottom) = (left + w, top + h);
545                      let border = window.scale_factor() * 5;
546                      let edge = crate::window::hit_test(
547                        (left, top, right, bottom),
548                        cx as _,
549                        cy as _,
550                        border,
551                        border,
552                      )
553                      .map(|d| d.to_gtk_edge())
554                      // we return `WindowEdge::__Unknown` to be ignored later.
555                      // we must return 8 or bigger, otherwise it will be the same as one of the other 7 variants of `WindowEdge` enum.
556                      .unwrap_or(WindowEdge::__Unknown(8));
557
558                      // Ignore the `__Unknown` variant so the window receives the click correctly if it is not on the edges.
559                      match edge {
560                        WindowEdge::__Unknown(_) => (),
561                        _ => window.begin_resize_drag_for_device(
562                          edge,
563                          &device,
564                          0,
565                          cx as i32,
566                          cy as i32,
567                          event.time(),
568                        ),
569                      }
570                    }
571                  }
572                }
573              }
574
575              glib::Propagation::Proceed
576            });
577
578            let tx_clone = event_tx.clone();
579            window.connect_delete_event(move |_, _| {
580              if let Err(e) = tx_clone.send(Event::WindowEvent {
581                window_id: RootWindowId(id),
582                event: WindowEvent::CloseRequested,
583              }) {
584                log::warn!("Failed to send window close event to event channel: {}", e);
585              }
586              glib::Propagation::Stop
587            });
588
589            let tx_clone = event_tx.clone();
590            window.connect_configure_event(move |window, event| {
591              let scale_factor = window.scale_factor();
592
593              let (x, y) = window
594                .window()
595                .map(|w| w.root_origin())
596                .unwrap_or_else(|| event.position());
597              if let Err(e) = tx_clone.send(Event::WindowEvent {
598                window_id: RootWindowId(id),
599                event: WindowEvent::Moved(
600                  LogicalPosition::new(x, y).to_physical(scale_factor as f64),
601                ),
602              }) {
603                log::warn!("Failed to send window moved event to event channel: {}", e);
604              }
605
606              let (w, h) = event.size();
607              if let Err(e) = tx_clone.send(Event::WindowEvent {
608                window_id: RootWindowId(id),
609                event: WindowEvent::Resized(
610                  LogicalSize::new(w, h).to_physical(scale_factor as f64),
611                ),
612              }) {
613                log::warn!(
614                  "Failed to send window resized event to event channel: {}",
615                  e
616                );
617              }
618              false
619            });
620
621            let tx_clone = event_tx.clone();
622            window.connect_focus_in_event(move |_, _| {
623              if let Err(e) = tx_clone.send(Event::WindowEvent {
624                window_id: RootWindowId(id),
625                event: WindowEvent::Focused(true),
626              }) {
627                log::warn!(
628                  "Failed to send window focus-in event to event channel: {}",
629                  e
630                );
631              }
632              glib::Propagation::Proceed
633            });
634
635            let tx_clone = event_tx.clone();
636            window.connect_focus_out_event(move |_, _| {
637              if let Err(e) = tx_clone.send(Event::WindowEvent {
638                window_id: RootWindowId(id),
639                event: WindowEvent::Focused(false),
640              }) {
641                log::warn!(
642                  "Failed to send window focus-out event to event channel: {}",
643                  e
644                );
645              }
646              glib::Propagation::Proceed
647            });
648
649            let tx_clone = event_tx.clone();
650            window.connect_destroy(move |_| {
651              if let Err(e) = tx_clone.send(Event::WindowEvent {
652                window_id: RootWindowId(id),
653                event: WindowEvent::Destroyed,
654              }) {
655                log::warn!(
656                  "Failed to send window destroyed event to event channel: {}",
657                  e
658                );
659              }
660            });
661
662            let tx_clone = event_tx.clone();
663            window.connect_enter_notify_event(move |_, _| {
664              if let Err(e) = tx_clone.send(Event::WindowEvent {
665                window_id: RootWindowId(id),
666                event: WindowEvent::CursorEntered {
667                  device_id: DEVICE_ID,
668                },
669              }) {
670                log::warn!(
671                  "Failed to send cursor entered event to event channel: {}",
672                  e
673                );
674              }
675              glib::Propagation::Proceed
676            });
677
678            let tx_clone = event_tx.clone();
679            window.connect_motion_notify_event(move |window, motion| {
680              if cursor_moved {
681                if let Some(cursor) = motion.device() {
682                  let scale_factor = window.scale_factor();
683                  let (_, x, y) = cursor.window_at_position();
684                  if let Err(e) = tx_clone.send(Event::WindowEvent {
685                    window_id: RootWindowId(id),
686                    event: WindowEvent::CursorMoved {
687                      position: LogicalPosition::new(x, y).to_physical(scale_factor as f64),
688                      device_id: DEVICE_ID,
689                      // this field is depracted so it is fine to pass empty state
690                      modifiers: ModifiersState::empty(),
691                    },
692                  }) {
693                    log::warn!("Failed to send cursor moved event to event channel: {}", e);
694                  }
695                }
696              }
697              glib::Propagation::Stop
698            });
699
700            let tx_clone = event_tx.clone();
701            window.connect_leave_notify_event(move |_, _| {
702              if let Err(e) = tx_clone.send(Event::WindowEvent {
703                window_id: RootWindowId(id),
704                event: WindowEvent::CursorLeft {
705                  device_id: DEVICE_ID,
706                },
707              }) {
708                log::warn!("Failed to send cursor left event to event channel: {}", e);
709              }
710              glib::Propagation::Proceed
711            });
712
713            let tx_clone = event_tx.clone();
714            window.connect_button_press_event(move |_, event| {
715              let button = event.button();
716              if let Err(e) = tx_clone.send(Event::WindowEvent {
717                window_id: RootWindowId(id),
718                event: WindowEvent::MouseInput {
719                  button: match button {
720                    1 => MouseButton::Left,
721                    2 => MouseButton::Middle,
722                    3 => MouseButton::Right,
723                    _ => MouseButton::Other(button as u16),
724                  },
725                  state: ElementState::Pressed,
726                  device_id: DEVICE_ID,
727                  // this field is depracted so it is fine to pass empty state
728                  modifiers: ModifiersState::empty(),
729                },
730              }) {
731                log::warn!(
732                  "Failed to send mouse input pressed event to event channel: {}",
733                  e
734                );
735              }
736              glib::Propagation::Stop
737            });
738
739            let tx_clone = event_tx.clone();
740            window.connect_button_release_event(move |_, event| {
741              let button = event.button();
742              if let Err(e) = tx_clone.send(Event::WindowEvent {
743                window_id: RootWindowId(id),
744                event: WindowEvent::MouseInput {
745                  button: match button {
746                    1 => MouseButton::Left,
747                    2 => MouseButton::Middle,
748                    3 => MouseButton::Right,
749                    _ => MouseButton::Other(button as u16),
750                  },
751                  state: ElementState::Released,
752                  device_id: DEVICE_ID,
753                  // this field is depracted so it is fine to pass empty state
754                  modifiers: ModifiersState::empty(),
755                },
756              }) {
757                log::warn!(
758                  "Failed to send mouse input released event to event channel: {}",
759                  e
760                );
761              }
762              glib::Propagation::Stop
763            });
764
765            let tx_clone = event_tx.clone();
766            window.connect_scroll_event(move |_, event| {
767              let (x, y) = event.delta();
768              if let Err(e) = tx_clone.send(Event::WindowEvent {
769                window_id: RootWindowId(id),
770                event: WindowEvent::MouseWheel {
771                  device_id: DEVICE_ID,
772                  delta: MouseScrollDelta::LineDelta(-x as f32, -y as f32),
773                  phase: match event.direction() {
774                    ScrollDirection::Smooth => TouchPhase::Moved,
775                    _ => TouchPhase::Ended,
776                  },
777                  modifiers: ModifiersState::empty(),
778                },
779              }) {
780                log::warn!("Failed to send scroll event to event channel: {}", e);
781              }
782              glib::Propagation::Proceed
783            });
784
785            let tx_clone = event_tx.clone();
786            let keyboard_handler = Rc::new(move |event_key: EventKey, element_state| {
787              // if we have a modifier lets send it
788              let mut mods = keyboard::get_modifiers(event_key.clone());
789              if !mods.is_empty() {
790                // if we release the modifier tell the world
791                if ElementState::Released == element_state {
792                  mods = ModifiersState::empty();
793                }
794
795                if let Err(e) = tx_clone.send(Event::WindowEvent {
796                  window_id: RootWindowId(id),
797                  event: WindowEvent::ModifiersChanged(mods),
798                }) {
799                  log::warn!(
800                    "Failed to send modifiers changed event to event channel: {}",
801                    e
802                  );
803                } else {
804                  // stop here we don't want to send the key event
805                  // as we emit the `ModifiersChanged`
806                  return glib::ControlFlow::Continue;
807                }
808              }
809
810              // todo: implement repeat?
811              let event = keyboard::make_key_event(&event_key, false, None, element_state);
812
813              if let Some(event) = event {
814                if let Err(e) = tx_clone.send(Event::WindowEvent {
815                  window_id: RootWindowId(id),
816                  event: WindowEvent::KeyboardInput {
817                    device_id: DEVICE_ID,
818                    event,
819                    is_synthetic: false,
820                  },
821                }) {
822                  log::warn!("Failed to send keyboard event to event channel: {}", e);
823                }
824              }
825              glib::ControlFlow::Continue
826            });
827
828            let tx_clone = event_tx.clone();
829            // TODO Add actual IME from system
830            let ime = gtk::IMContextSimple::default();
831            ime.set_client_window(window.window().as_ref());
832            ime.focus_in();
833            ime.connect_commit(move |_, s| {
834              if let Err(e) = tx_clone.send(Event::WindowEvent {
835                window_id: RootWindowId(id),
836                event: WindowEvent::ReceivedImeText(s.to_string()),
837              }) {
838                log::warn!(
839                  "Failed to send received IME text event to event channel: {}",
840                  e
841                );
842              }
843            });
844
845            let handler = keyboard_handler.clone();
846            window.connect_key_press_event(move |_, event_key| {
847              handler(event_key.to_owned(), ElementState::Pressed);
848              ime.filter_keypress(event_key);
849
850              glib::Propagation::Proceed
851            });
852
853            let handler = keyboard_handler.clone();
854            window.connect_key_release_event(move |_, event_key| {
855              handler(event_key.to_owned(), ElementState::Released);
856              glib::Propagation::Proceed
857            });
858
859            let tx_clone = event_tx.clone();
860            window.connect_window_state_event(move |window, event| {
861              let state = event.changed_mask();
862              if state.contains(WindowState::ICONIFIED) || state.contains(WindowState::MAXIMIZED) {
863                let scale_factor = window.scale_factor();
864
865                let (x, y) = window.position();
866                if let Err(e) = tx_clone.send(Event::WindowEvent {
867                  window_id: RootWindowId(id),
868                  event: WindowEvent::Moved(
869                    LogicalPosition::new(x, y).to_physical(scale_factor as f64),
870                  ),
871                }) {
872                  log::warn!("Failed to send window moved event to event channel: {}", e);
873                }
874
875                let (w, h) = window.size();
876                if let Err(e) = tx_clone.send(Event::WindowEvent {
877                  window_id: RootWindowId(id),
878                  event: WindowEvent::Resized(
879                    LogicalSize::new(w, h).to_physical(scale_factor as f64),
880                  ),
881                }) {
882                  log::warn!(
883                    "Failed to send window resized event to event channel: {}",
884                    e
885                  );
886                }
887              }
888              glib::Propagation::Proceed
889            });
890
891            // Receive draw events of the window.
892            let draw_clone = draw_tx.clone();
893            window.connect_draw(move |window, cr| {
894              if let Err(e) = draw_clone.send(id) {
895                log::warn!("Failed to send redraw event to event channel: {}", e);
896              }
897
898              if transparent {
899                let background_color = unsafe {
900                  window
901                    .data::<Option<crate::window::RGBA>>("background_color")
902                    .and_then(|c| c.as_ref().clone())
903                };
904
905                let rgba = background_color
906                  .map(|(r, g, b, a)| (r as f64, g as f64, b as f64, a as f64 / 255.0))
907                  .unwrap_or((0., 0., 0., 0.));
908
909                let rect = window
910                  .child()
911                  .map(|c| c.allocation())
912                  .unwrap_or_else(|| window.allocation());
913
914                cr.rectangle(
915                  rect.x() as _,
916                  rect.y() as _,
917                  rect.width() as _,
918                  rect.height() as _,
919                );
920                cr.set_source_rgba(rgba.0, rgba.1, rgba.2, rgba.3);
921                cr.set_operator(cairo::Operator::Source);
922                let _ = cr.fill();
923                cr.set_operator(cairo::Operator::Over);
924              }
925
926              glib::Propagation::Proceed
927            });
928          }
929        }
930      } else if id == WindowId::dummy() {
931        match request {
932          WindowRequest::ProgressBarState(state) => {
933            taskbar.update(state);
934          }
935          WindowRequest::BadgeCount(count, desktop_filename) => {
936            taskbar.update_count(count, desktop_filename);
937          }
938          WindowRequest::SetTheme(theme) => {
939            if let Some(settings) = Settings::default() {
940              match theme {
941                Some(Theme::Dark) => settings.set_gtk_application_prefer_dark_theme(true),
942                Some(Theme::Light) | None => settings.set_gtk_application_prefer_dark_theme(false),
943              }
944            }
945          }
946          _ => unreachable!(),
947        }
948      }
949      glib::ControlFlow::Continue
950    });
951
952    // Create event loop itself.
953    let event_loop = Self {
954      window_target: RootELW {
955        p: window_target,
956        _marker: std::marker::PhantomData,
957      },
958      user_event_tx,
959      events: event_rx,
960      draws: draw_rx,
961      run_device_thread,
962    };
963
964    Ok(event_loop)
965  }

967  #[inline]
968  pub fn run<F>(mut self, callback: F) -> !
969  where
970    F: FnMut(Event<'_, T>, &RootELW<T>, &mut ControlFlow) + 'static,
971  {
972    let exit_code = self.run_return(callback);
973    process::exit(exit_code)
974  }
975
976  /// This is the core event loop logic. It basically loops on `gtk_main_iteration` and processes one
977  /// event along with that iteration. Depends on current control flow and what it should do, an
978  /// event state is defined. The whole state flow chart runs like following:
979  ///
980  /// ```ignore
981  ///                                   Poll/Wait/WaitUntil
982  ///       +-------------------------------------------------------------------------+
983  ///       |                                                                         |
984  ///       |                   Receiving event from event channel                    |   Receiving event from draw channel
985  ///       |                               +-------+                                 |   +---+
986  ///       v                               v       |                                 |   v   |
987  /// +----------+  Poll/Wait/WaitUntil   +------------+  Poll/Wait/WaitUntil   +-----------+ |
988  /// | NewStart | ---------------------> | EventQueue | ---------------------> | DrawQueue | |
989  /// +----------+                        +------------+                        +-----------+ |
990  ///       |ExitWithCode                        |ExitWithCode            ExitWithCode|   |   |
991  ///       +------------------------------------+------------------------------------+   +---+
992  ///                                            |
993  ///                                            v
994  ///                                    +---------------+
995  ///                                    | LoopDestroyed |
996  ///                                    +---------------+
997  /// ```
998  ///
999  /// There are a dew notibale event will sent to callback when state is transisted:
1000  /// - On any state moves to `LoopDestroyed`, a `LoopDestroyed` event is sent.
1001  /// - On `NewStart` to `EventQueue`, a `NewEvents` with corresponding `StartCause` depends on
1002  /// current control flow is sent.
1003  /// - On `EventQueue` to `DrawQueue`, a `MainEventsCleared` event is sent.
1004  /// - On `DrawQueue` back to `NewStart`, a `RedrawEventsCleared` event is sent.
1005  pub(crate) fn run_return<F>(&mut self, mut callback: F) -> i32
1006  where
1007    F: FnMut(Event<'_, T>, &RootELW<T>, &mut ControlFlow),
1008  {
1009    enum EventState {
1010      NewStart,
1011      EventQueue,
1012      DrawQueue,
1013    }
1014
1015    let context = MainContext::default();
1016    let run_device_thread = self.run_device_thread.clone();
1017
1018    context
1019      .with_thread_default(|| {
1020        let mut control_flow = ControlFlow::default();
1021        let window_target = &self.window_target;
1022        let events = &self.events;
1023        let draws = &self.draws;
1024
1025        window_target.p.app.activate();
1026
1027        let mut state = EventState::NewStart;
1028        let exit_code = loop {
1029          let mut blocking = false;
1030          match state {
1031            EventState::NewStart => match control_flow {
1032              ControlFlow::ExitWithCode(code) => {
1033                callback(Event::LoopDestroyed, window_target, &mut control_flow);
1034                break code;
1035              }
1036              ControlFlow::Wait => {
1037                if !events.is_empty() {
1038                  callback(
1039                    Event::NewEvents(StartCause::WaitCancelled {
1040                      start: Instant::now(),
1041                      requested_resume: None,
1042                    }),
1043                    window_target,
1044                    &mut control_flow,
1045                  );
1046                  state = EventState::EventQueue;
1047                } else {
1048                  blocking = true;
1049                }
1050              }
1051              ControlFlow::WaitUntil(requested_resume) => {
1052                let start = Instant::now();
1053                if start >= requested_resume {
1054                  callback(
1055                    Event::NewEvents(StartCause::ResumeTimeReached {
1056                      start,
1057                      requested_resume,
1058                    }),
1059                    window_target,
1060                    &mut control_flow,
1061                  );
1062                  state = EventState::EventQueue;
1063                } else if !events.is_empty() {
1064                  callback(
1065                    Event::NewEvents(StartCause::WaitCancelled {
1066                      start,
1067                      requested_resume: Some(requested_resume),
1068                    }),
1069                    window_target,
1070                    &mut control_flow,
1071                  );
1072                  state = EventState::EventQueue;
1073                } else {
1074                  blocking = true;
1075                }
1076              }
1077              _ => {
1078                callback(
1079                  Event::NewEvents(StartCause::Poll),
1080                  window_target,
1081                  &mut control_flow,
1082                );
1083                state = EventState::EventQueue;
1084              }
1085            },
1086            EventState::EventQueue => match control_flow {
1087              ControlFlow::ExitWithCode(code) => {
1088                callback(Event::LoopDestroyed, window_target, &mut control_flow);
1089                break (code);
1090              }
1091              _ => match events.try_recv() {
1092                Ok(event) => match event {
1093                  Event::LoopDestroyed => control_flow = ControlFlow::ExitWithCode(1),
1094                  _ => callback(event, window_target, &mut control_flow),
1095                },
1096                Err(_) => {
1097                  callback(Event::MainEventsCleared, window_target, &mut control_flow);
1098                  state = EventState::DrawQueue;
1099                }
1100              },
1101            },
1102            EventState::DrawQueue => match control_flow {
1103              ControlFlow::ExitWithCode(code) => {
1104                callback(Event::LoopDestroyed, window_target, &mut control_flow);
1105                break code;
1106              }
1107              _ => {
1108                if let Ok(id) = draws.try_recv() {
1109                  callback(
1110                    Event::RedrawRequested(RootWindowId(id)),
1111                    window_target,
1112                    &mut control_flow,
1113                  );
1114                }
1115                callback(Event::RedrawEventsCleared, window_target, &mut control_flow);
1116                state = EventState::NewStart;
1117              }
1118            },
1119          }
1120          gtk::main_iteration_do(blocking);
1121        };
1122        if let Some(run_device_thread) = run_device_thread {
1123          run_device_thread.store(false, Ordering::Relaxed);
1124        }
1125        exit_code
1126      })
1127      .unwrap_or(1)
1128  }

1143/// Used to send custom events to `EventLoop`.
1144#[derive(Debug)]
1145pub struct EventLoopProxy<T: 'static> {
1146  user_event_tx: crossbeam_channel::Sender<Event<'static, T>>,
1147}
1148
1149impl<T: 'static> Clone for EventLoopProxy<T> {
1150  fn clone(&self) -> Self {
1151    Self {
1152      user_event_tx: self.user_event_tx.clone(),
1153    }
1154  }
1155}
1156
1157impl<T: 'static> EventLoopProxy<T> {
1158  /// Send an event to the `EventLoop` from which this proxy was created. This emits a
1159  /// `UserEvent(event)` event in the event loop, where `event` is the value passed to this
1160  /// function.
1161  ///
1162  /// Returns an `Err` if the associated `EventLoop` no longer exists.
1163  pub fn send_event(&self, event: T) -> Result<(), EventLoopClosed<T>> {
1164    self
1165      .user_event_tx
1166      .send(Event::UserEvent(event))
1167      .map_err(|SendError(event)| {
1168        if let Event::UserEvent(error) = event {
1169          EventLoopClosed(error)
1170        } else {
1171          unreachable!();
1172        }
1173      })?;
1174
1175    let context = MainContext::default();
1176    context.wakeup();
1177
1178    Ok(())
1179  }
1180}

1188  );
