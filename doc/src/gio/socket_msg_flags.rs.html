57unsafe impl<'a> FromValue<'a> for SocketMsgFlags {

62    }
63}

65impl ToValue for SocketMsgFlags {
66    fn to_value(&self) -> glib::Value {
67        let mut value = glib::Value::for_value_type::<Self>();
68        unsafe {
