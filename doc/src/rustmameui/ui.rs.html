1//! This module contains the Dioxus UI code for the application.
2//! It defines the main application component (`App`), handles state management
3//! using signals and memos, renders the tabbed interface for games, favourites,
4//! settings, and about sections, and manages user interactions like selecting
5//! games, launching games, adding/removing favourites, updating settings,
6//! and refreshing the game list.
7

9use crate::rustmameuiconfig::Config;
10use crate::game::Game;
11use rfd::FileDialog; // Used for native file/folder dialogs
12use rust_i18n::t; // Macro for internationalization (i18n)

14// Define external assets
15const TABS_CSS: Asset = asset!("/assets/tabs.css");
16const ABOUT: Asset = asset!("/assets/about.svg");

18// Application name constant
19const APP_NAME: &str = "MAMEUI";
20// Batch size for processing ROMs (e.g., verification)
21const BATCH_SIZE: usize = 100;

32/// The main entry point for drawing and launching the application UI.
33pub fn draw() {
34    #[cfg(feature = "desktop")]
35    fn launch_app() {
36        let window = dioxus::desktop::tao::window::WindowBuilder::new()
37            .with_resizable(true)
38            .with_inner_size(dioxus::desktop::tao::dpi::LogicalSize::new(1024.0, 860.0))
39            .with_title::<&str>(APP_NAME);
40        dioxus::LaunchBuilder::new()
41            .with_cfg(dioxus::desktop::Config::new().with_window(window).with_menu(None))
42            .launch(App);
43    }
44
45    #[cfg(not(feature = "desktop"))]
46    fn launch_app() {
47        dioxus::launch(App);
48    }
49
50    launch_app();
51}

53/// Checks if a native file dialog utility (zenity or kdialog) is installed.
54fn check_dialog_utility() -> Result<(), String> {
55    let mut found = false;
56    if which::which("zenity").is_ok() {
57        found = true;
58    }
59    if which::which("kdialog").is_ok() {
60        found = true;
61    }
62    if !found {
63        return Err(t!("error_no_dialog_utility").to_string());
64    }
65    Ok(())
66}

68/// Component for the Games tab content.
69#[component]
70fn GamesTab(
71    games: Signal<Vec<Game>>,
72    selected_game: Signal<(String, String, bool)>,
73    game_filter: Signal<String>,
74    snap_data: Signal<String>,
75    mame_executable: Signal<std::path::PathBuf>,
76    roms_path: Signal<std::path::PathBuf>,
77    snap_file: Signal<std::path::PathBuf>,
78    favourites: Signal<Vec<Game>>,
79    show_context_menu: Signal<bool>,
80    menu_x: Signal<i32>,
81    menu_y: Signal<i32>,
82    selected_game_popup_menu_label: Signal<String>,
83    selected_game_popup_menu_current_label: ReadOnlySignal<String>,
84    selected_game_is_favourite: ReadOnlySignal<bool>,
85) -> Element {
86    let filtered_games = use_memo(move || {
87        let filter = game_filter();
88        let all_games = games();
89        all_games
90            .iter()
91            .filter(|item| item.description().to_lowercase().contains(&filter.to_lowercase()))
92            .cloned()
93            .collect::<Vec<Game>>()
94    });
95
96    let handle_context_menu = move |event: Event<MouseData>| {
97        event.stop_propagation();
98        event.prevent_default();
99        let (x, y) = (event.coordinates().client().x as i32, event.coordinates().client().y as i32);
100        menu_x.set(x);
101        menu_y.set(y);
102        show_context_menu.set(true);
103    };
104
105    rsx! {
106        div {
107            class: "tab-panel active",
108            // Header
109            h2 { {t!("all_games")} }
110            p {
111                margin_bottom: "1.5em",
112                {t!("double_click_on_a_game_to_launch_right_click_for_more_options")}
113            }
114            // Content Layout
115            div {
116                class: "row",
117                // Left Column: Game List and Filter
118                div {
119                    class: "column",
120                    // Game List
121                    select {
122                        style: "max-width: 500px;",
123                        resize: "both",
124                        autocomplete: "on",
125                        size: 20,
126                        oncontextmenu: handle_context_menu,
127                        onchange: move |event| {
128                            let rom = event.value().clone();
129                            let description = games.read().iter()
130                                .find(|game| game.rom().clone() == rom)
131                                .map(|game| game.description().clone())
132                                .unwrap_or_default();
133                            let snap = games.read().iter()
134                                .find(|game| game.rom().clone() == rom)
135                                .map(|game| game.snap())
136                                .unwrap_or_default();
137                            snap_data.set(Game::new(&rom, &description, snap).get_snap(&snap_file().display().to_string()).unwrap());
138                            selected_game.set((rom, description, snap));
139                        },
140                        ondoubleclick: move |_| {
141                            let (rom, description, snap) = selected_game();
142                            let game = Game::new(&rom, &description, snap);
143                            let _ = game.launch(&mame_executable(), &roms_path());
144                        },
145                        onkeypress: move |ev: Event<KeyboardData>| {
146                            if ev.key() == Key::Enter {
147                                let (rom, description, snap) = selected_game();
148                                let game = Game::new(&rom, &description, snap);
149                                let _ = game.launch(&mame_executable(), &roms_path());
150                            }
151                        },
152                        for game in filtered_games() {
153                            option {
154                                value: game.rom().clone(),
155                                {game.description().clone()}
156                            }
157                        }
158                    }
159                    // Context Menu
160                    {
161                        if show_context_menu() {
162                            if selected_game_is_favourite() {
163                                selected_game_popup_menu_label.set(t!("remove_from_favourites").to_string());
164                            } else {
165                                selected_game_popup_menu_label.set(t!("add_to_favourites").to_string());
166                            }
167                            rsx! {
168                                div {
169                                    class: "popup",
170                                    style: "left: {menu_x()}px; top: {menu_y()}px;",
171                                    div {
172                                        style: "cursor: pointer; padding: 10px;",
173                                        onclick: move |_| {
174                                            // Use Config::new() to be sure to use the last configuration saved on disk
175                                            let config = Config::new().unwrap();
176                                            let (rom, description, snap) = selected_game();
177                                            let game = Game::new(&rom, &description, snap);
178                                            let _ = game.launch(&config.mame_executable, &config.roms_path);
179                                            show_context_menu.set(false);
180                                        },
181                                        {t!("launch_game")}
182                                    }
183                                    div {
184                                        style: "cursor: pointer; padding: 10px;",
185                                        onclick: move |_| {
186                                            let config = Config::new().unwrap();
187                                            let (rom, description, snap) = selected_game();
188                                            let favourite = Game::new(&rom, &description, snap);
189                                            let new_favourites = if selected_game_is_favourite() {
190                                                crate::games::remove_favourite(&config, &mut favourites(), &favourite)
191                                            } else {
192                                                crate::games::add_favourite(&config, &mut favourites(), &favourite)
193                                            };
194                                            favourites.set(new_favourites);
195                                            show_context_menu.set(false);
196                                        },
197                                        {selected_game_popup_menu_current_label()}
198                                    }
199                                }
200                            }
201                        } else {
202                            rsx! {}
203                        }
204                    }
205                    // Filter Input
206                    div {
207                        div {
208                            class: "column",
209                            input {
210                                r#type: "text",
211                                size: 50,
212                                oninput: move |evt| {
213                                    game_filter.set(evt.value());
214                                }
215                            }
216                            label {
217                                {t!("type_the_name_of_a_game_to_search_for_it")}
218                            }
219                        }
220                    }
221                }
222                // Right Column: Snapshot
223                div {
224                    class: "snap",
225                    img {
226                        src: "{snap_data()}",
227                        width: 500,
228                    }
229                }
230            }
231        }
232    }
233}

430                }

432            // ROMs Directory Path
433            div {
434                class: "setting",
435                label { {t!("roms_directory")} }
436                input {
437                    size: 50,
438                    value: "{roms_path().display().to_string()}",
439                    onchange: move |evt| {
440                        roms_path.set(std::path::PathBuf::from(evt.value()));
441                    }
442                }
443                button {
444                    onclick: move |_| {
445                        match check_dialog_utility() {
446                            Ok(_) => {
447                                if let Some(directory) = FileDialog::new()
448                                    .add_filter(t!("all_files"), &["*"])
449                                    .pick_folder() {
450                                    roms_path.set(directory);
451                                }
452                            },
453                            Err(e) => {
454                                panic!("{}", e);
455                            }
456                        }
457                    },
458                    {t!("browse")}
459                }
460            }
461            // Snapshots Zip File Path
462            div {
463                class: "setting",
464                label { {t!("snaps_zip_not_7z_file_path")} }
465                input {
466                    size: 50,
467                    value: "{snap_file().display().to_string()}",
468                    onchange: move |evt| {
469                        snap_file.set(std::path::PathBuf::from(evt.value()));
470                    }
471                }
472                button {
473                    onclick: move |_| {
474                        match check_dialog_utility() {
475                            Ok(_) => {
476                                if let Some(file) = FileDialog::new()
477                                    .add_filter(t!("zip_file"), &["*.zip"])
478                                    .pick_file() {
479                                    snap_file.set(file);
480                                }
481                            },
482                            Err(e) => {
483                                panic!("{}", e);
484                            }
485                        }
486                    },
487                    "Browse"
488                }
489            }
490            // Action Buttons
491            div {
492                div {
493                    class: "row",
494                    button {
495                        onclick: move |_| {
496                            let mut config = Config::new().unwrap();
497                            config.mame_executable = mame_executable();
498                            config.roms_path = roms_path();
499                            config.snap_file = snap_file();
500                            match config.save() {
501                                Ok(_) => {
502                                    status.set(t!("the_settings_have_been_saved_correctly").into());
503                                },
504                                Err(e) => {
505                                    status.set(t!("error_while_saving_settings.error", error = e.to_string()).to_string());
506                                }
507                            }
508                        },
509                        "Save settings"
510                    }
511                    button {
512                        class: "action-button",
513                        onclick: move |_| async move {
514                            let config_third_clone = Config::new().unwrap();
515                            status.set(t!("reading_the_roms_list_from_mame").into());
516                            tokio::time::sleep(std::time::Duration::from_millis(10)).await;
517                            let xml_string = crate::games::get_xml_roms(&config_third_clone)
518                                .unwrap_or_else(|_| { panic!("{}",  t!("cannot_get_xml_data_from_mame").to_string() )});
519                            status.set("Parsing the roms list to get valid roms...".into());
520                            tokio::time::sleep(std::time::Duration::from_millis(10)).await;
521                            let (roms, descriptions) = crate::games::get_all_roms(&config_third_clone, &xml_string)
522                                .unwrap_or_else(|_| { panic!("{}", t!("cannot_read_all_roms_and_their_descriptions").to_string()) });
523                            let mut all_games: Vec<Game> = Vec::new();
524                            status.set(t!("verifying_working_roms").into());
525                            tokio::time::sleep(std::time::Duration::from_millis(10)).await;
526                            let total_roms = roms.len();
527                            let mut working_roms = Vec::with_capacity(total_roms);
528                            let mut working_snaps = Vec::with_capacity(total_roms);
529                            let num_batches = total_roms.div_ceil(BATCH_SIZE);
530                            for batch_index in 0..num_batches {
531                                let start_index = batch_index * BATCH_SIZE;
532                                let end_index = usize::min((batch_index + 1) * BATCH_SIZE, total_roms);
533                                status.set(t!("verifying_roms.from.to.of.total", from = (start_index + 1), to = end_index, total = total_roms).into());
534                                tokio::time::sleep(std::time::Duration::from_millis(10)).await;
535                                let current_batch = &roms[start_index..end_index];
536                                let roms_batch_results = crate::games::verify_batch_roms(&config_third_clone, current_batch);
537                                working_roms.extend(roms_batch_results);
538                                status.set(t!("verifying_snaps.from.to.of.total", from = (start_index + 1), to = end_index, total = total_roms).into());
539                                tokio::time::sleep(std::time::Duration::from_millis(10)).await;
540                                let snaps_batch_results = crate::games::verify_batch_snaps(&config_third_clone, current_batch);
541                                working_snaps.extend(snaps_batch_results);
542                            }
543                            if !roms.is_empty() {
544                                for (i, rom) in roms.iter().enumerate() {
545                                    if working_roms.get(i).copied().unwrap_or(false) {
546                                        let game = Game::new(rom, descriptions.get(i).unwrap(), working_snaps.get(i).copied().unwrap_or(false));
547                                        all_games.push(game);
548                                    }
549                                }
550                            }
551                            status.set(t!("saving_all_games").into());
552                            tokio::time::sleep(std::time::Duration::from_millis(10)).await;
553                            match crate::games::save(&config_third_clone, &all_games) {
554                                Ok(_) => {},
555                                Err(e) => {
556                                    panic!("{}", t!("error_while_saving_games_list.error", error = e.to_string()).to_string());
557                                }
558                            };
559                            status.set(t!("complete").into());
560                            games.set(all_games);
561                        },
562                        {t!("refresh_games_list")}
563                    }
564                }
565                div { {status()} }
566            }
567        }
568    }
569}

612/// The main application component.
613#[component]
614fn App() -> Element {
615    let app_version = env!("CARGO_PKG_VERSION");
616    let mut selected_tab = use_signal(|| Tab::Games);
617
618    let config = match Config::new() {
619        Ok(c) => c,
620        Err(e) => panic!("{}", t!("error.error", error = e.to_string()).to_string()),
621    };
622
623    let config_clone = config.clone();
624    let mut mame_executable = use_signal(|| config_clone.mame_executable);
625    let mut snap_file = use_signal(|| config_clone.snap_file);
626    let mut roms_path = use_signal(|| config_clone.roms_path);
627
628    let mut snap_data = use_signal(|| "".to_string());
629    let mut favourite_snap_data = use_signal(|| "".to_string());
630
631    let mut games = use_signal(|| crate::games::load(&config).unwrap());
632    let mut favourites = use_signal(|| crate::games::load_favourites(&config).unwrap());
633
634    let mut selected_game = use_signal(|| (String::new(), String::new(), false));
635    let mut selected_favourite = use_signal(|| (String::new(), String::new(), false));
636
637    let mut game_filter = use_signal(String::new);
638    let mut favourite_filter = use_signal(String::new);
639
640    let filtered_games = use_memo(move || {
641        let filter = game_filter();
642        let all_games = games();
643        all_games
644            .iter()
645            .filter(|item| item.description().to_lowercase().contains(&filter.to_lowercase()))
646            .cloned()
647            .collect::<Vec<Game>>()
648    });
649
650    let filtered_favourites = use_memo(move || {
651        let filter = favourite_filter();
652        let all_favourites = favourites();
653        all_favourites
654            .iter()
655            .filter(|item| item.description().to_lowercase().contains(&filter.to_lowercase()))
656            .cloned()
657            .collect::<Vec<Game>>()
658    });
659
660    let mut status = use_signal(|| String::from(""));
661
662    let mut selected_game_popup_menu_label = use_signal(|| "".to_string());
663    let selected_game_popup_menu_current_label = use_memo(move || selected_game_popup_menu_label());
664
665    let selected_game_is_favourite = use_memo(move || {
666        let (rom, _, _) = selected_game();
667        let mut found = false;
668        for favourite in favourites() {
669            if favourite.rom() == rom {
670                found = true;
671            }
672        }
673        found
674    });
675
676    let mut show_context_menu = use_signal(|| false);
677    let mut menu_x = use_signal(|| 0_i32);
678    let mut menu_y = use_signal(|| 0_i32);
679
680    let hide_menu = move |_| {
681        show_context_menu.set(false);
682    };
683
684    rsx! {
685        document::Link { rel: "stylesheet", href: TABS_CSS }
686        div {
687            class: "container",
688            // Tabs Header
689            div {
690                class: "tabs-header",
691                button {
692                    class: if selected_tab() == Tab::Games { "tab-button active" } else { "tab-button" },
693                    onclick: move |_| selected_tab.set(Tab::Games),
694                    {t!("games")}
695                }
696                button {
697                    class: if selected_tab() == Tab::Favourites { "tab-button active" } else { "tab-button" },
698                    onclick: move |_| selected_tab.set(Tab::Favourites),
699                    {t!("favourites")}
700                }
701                button {
702                    class: if selected_tab() == Tab::Settings { "tab-button active" } else { "tab-button" },
703                    onclick: move |_| selected_tab.set(Tab::Settings),
704                    {t!("settings")}
705                }
706                button {
707                    class: if selected_tab() == Tab::About { "tab-button active" } else { "tab-button" },
708                    onclick: move |_| selected_tab.set(Tab::About),
709                    {t!("about")}
710                }
711            }
712            // Tab Content
713            div {
714                onclick: hide_menu,
715                class: "tab-content",
716                if selected_tab() == Tab::Games {
717                    GamesTab {
718                        games,
719                        selected_game,
720                        game_filter,
721                        snap_data,
722                        mame_executable,
723                        roms_path,
724                        snap_file,
725                        favourites,
726                        show_context_menu,
727                        menu_x,
728                        menu_y,
729                        selected_game_popup_menu_label,
730                        selected_game_popup_menu_current_label,
731                        selected_game_is_favourite,
732                    }
733                }
734                if selected_tab() == Tab::Favourites {
735                    FavouritesTab {
736                        favourites,
737                        selected_favourite,
738                        favourite_filter,
739                        favourite_snap_data,
740                        mame_executable,
741                        roms_path,
742                        snap_file,
743                        show_context_menu,
744                        menu_x,
745                        menu_y,
746                    }
747                }
748                if selected_tab() == Tab::Settings {
749                    SettingsTab {
750                        mame_executable,
751                        roms_path,
752                        snap_file,
753                        status,
754                        games,
755                    }
756                }
757                if selected_tab() == Tab::About {
758                    AboutTab {}
759                }
760            }
761        }
762    }
