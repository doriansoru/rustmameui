16enum ThreadMessage {
17    RegisterHotKey(HotKey, Sender<crate::Result<()>>),
18    RegisterHotKeys(Vec<HotKey>, Sender<crate::Result<()>>),
19    UnRegisterHotKey(HotKey, Sender<crate::Result<()>>),
20    UnRegisterHotKeys(Vec<HotKey>, Sender<crate::Result<()>>),
21    DropThread,
22}
23
24pub struct GlobalHotKeyManager {
25    thread_tx: Sender<ThreadMessage>,
26}
27
28impl GlobalHotKeyManager {
29    pub fn new() -> crate::Result<Self> {
30        let (thread_tx, thread_rx) = unbounded();
31        std::thread::spawn(|| events_processor(thread_rx));
32        Ok(Self { thread_tx })
33    }
34
35    pub fn register(&self, hotkey: HotKey) -> crate::Result<()> {
36        let (tx, rx) = crossbeam_channel::bounded(1);
37        let _ = self
38            .thread_tx
39            .send(ThreadMessage::RegisterHotKey(hotkey, tx));
40
41        if let Ok(result) = rx.recv() {
42            result?;
43        }
44
45        Ok(())
46    }
47
48    pub fn unregister(&self, hotkey: HotKey) -> crate::Result<()> {
49        let (tx, rx) = crossbeam_channel::bounded(1);
50        let _ = self
51            .thread_tx
52            .send(ThreadMessage::UnRegisterHotKey(hotkey, tx));
53
54        if let Ok(result) = rx.recv() {
55            result?;
56        }
57
58        Ok(())
59    }
60
61    pub fn register_all(&self, hotkeys: &[HotKey]) -> crate::Result<()> {
62        let (tx, rx) = crossbeam_channel::bounded(1);
63        let _ = self
64            .thread_tx
65            .send(ThreadMessage::RegisterHotKeys(hotkeys.to_vec(), tx));
66
67        if let Ok(result) = rx.recv() {
68            result?;
69        }
70
71        Ok(())
72    }
73
74    pub fn unregister_all(&self, hotkeys: &[HotKey]) -> crate::Result<()> {
75        let (tx, rx) = crossbeam_channel::bounded(1);
76        let _ = self
77            .thread_tx
78            .send(ThreadMessage::UnRegisterHotKeys(hotkeys.to_vec(), tx));
79
80        if let Ok(result) = rx.recv() {
81            result?;
82        }
83
84        Ok(())
85    }
86}
87
88impl Drop for GlobalHotKeyManager {
89    fn drop(&mut self) {
90        let _ = self.thread_tx.send(ThreadMessage::DropThread);
91    }
92}

94// XGrabKey works only with the exact state (modifiers)
95// and since X11 considers NumLock, ScrollLock and CapsLock a modifier when it is ON,
96// we also need to register our shortcut combined with these extra modifiers as well
97const IGNORED_MODS: [u32; 4] = [
98    0,              // modifier only
99    xlib::Mod2Mask, // NumLock
100    xlib::LockMask, // CapsLock
101    xlib::Mod2Mask | xlib::LockMask,
102];
103
104#[inline]
105fn register_hotkey(
106    xlib: &Xlib,
107    display: *mut _XDisplay,
108    root: u64,
109    hotkeys: &mut BTreeMap<u32, Vec<(u32, u32, bool)>>,
110    hotkey: HotKey,
111) -> crate::Result<()> {
112    let (modifiers, key) = (
113        modifiers_to_x11_mods(hotkey.mods),
114        keycode_to_x11_scancode(hotkey.key),
115    );
116
117    if let Some(key) = key {
118        let keycode = unsafe { (xlib.XKeysymToKeycode)(display, key as _) };
119
120        for m in IGNORED_MODS {
121            let result = unsafe {
122                (xlib.XGrabKey)(
123                    display,
124                    keycode as _,
125                    modifiers | m,
126                    root,
127                    0,
128                    xlib::GrabModeAsync,
129                    xlib::GrabModeAsync,
130                )
131            };
132
133            if result == xlib::BadAccess as _ {
134                for m in IGNORED_MODS {
135                    unsafe { (xlib.XUngrabKey)(display, keycode as _, modifiers | m, root) };
136                }
137
138                return Err(crate::Error::AlreadyRegistered(hotkey));
139            }
140        }
141
142        let entry = hotkeys.entry(keycode as _).or_default();
143        match entry.iter().find(|e| e.1 == modifiers) {
144            None => {
145                entry.push((hotkey.id(), modifiers, false));
146                Ok(())
147            }
148            Some(_) => Err(crate::Error::AlreadyRegistered(hotkey)),
149        }
150    } else {
151        Err(crate::Error::FailedToRegister(format!(
152            "Unable to register accelerator (unknown scancode for this key: {}).",
153            hotkey.key
154        )))
155    }
156}

158#[inline]
159fn unregister_hotkey(
160    xlib: &Xlib,
161    display: *mut _XDisplay,
162    root: u64,
163    hotkeys: &mut BTreeMap<u32, Vec<(u32, u32, bool)>>,
164    hotkey: HotKey,
165) -> crate::Result<()> {
166    let (modifiers, key) = (
167        modifiers_to_x11_mods(hotkey.mods),
168        keycode_to_x11_scancode(hotkey.key),
169    );
170
171    if let Some(key) = key {
172        let keycode = unsafe { (xlib.XKeysymToKeycode)(display, key as _) };
173
174        for m in IGNORED_MODS {
175            unsafe { (xlib.XUngrabKey)(display, keycode as _, modifiers | m, root) };
176        }
177
178        let entry = hotkeys.entry(keycode as _).or_default();
179        entry.retain(|k| k.1 != modifiers);
180        Ok(())
181    } else {
182        Err(crate::Error::FailedToUnRegister(hotkey))
183    }
184}

186fn events_processor(thread_rx: Receiver<ThreadMessage>) {
187    //                           key    id,  mods, pressed
188    let mut hotkeys = BTreeMap::<u32, Vec<(u32, u32, bool)>>::new();
189    if let Ok(xlib) = xlib::Xlib::open() {
190        unsafe {
191            let display = (xlib.XOpenDisplay)(ptr::null());
192            let root = (xlib.XDefaultRootWindow)(display);
193
194            // Only trigger key release at end of repeated keys
195            let mut supported_rtrn: i32 = 0;
196            (xlib.XkbSetDetectableAutoRepeat)(display, 1, &mut supported_rtrn);
197
198            (xlib.XSelectInput)(display, root, xlib::KeyPressMask);
199            let mut event: xlib::XEvent = std::mem::zeroed();
200
201            loop {
202                // Always service all pending events to avoid a queue of events from building up.
203                while (xlib.XPending)(display) > 0 {
204                    (xlib.XNextEvent)(display, &mut event);
205                    match event.get_type() {
206                        e @ xlib::KeyPress | e @ xlib::KeyRelease => {
207                            let keycode = event.key.keycode;
208                            // X11 sends masks for Lock keys also and we only care about the 4 below
209                            let event_mods = event.key.state
210                                & (xlib::ControlMask
211                                    | xlib::ShiftMask
212                                    | xlib::Mod4Mask
213                                    | xlib::Mod1Mask);
214
215                            if let Some(entry) = hotkeys.get_mut(&keycode) {
216                                match e {
217                                    xlib::KeyPress => {
218                                        for (id, mods, pressed) in entry {
219                                            if event_mods == *mods && !*pressed {
220                                                GlobalHotKeyEvent::send(GlobalHotKeyEvent {
221                                                    id: *id,
222                                                    state: crate::HotKeyState::Pressed,
223                                                });
224                                                *pressed = true;
225                                            }
226                                        }
227                                    }
228                                    xlib::KeyRelease => {
229                                        for (id, _, pressed) in entry {
230                                            if *pressed {
231                                                GlobalHotKeyEvent::send(GlobalHotKeyEvent {
232                                                    id: *id,
233                                                    state: crate::HotKeyState::Released,
234                                                });
235                                                *pressed = false;
236                                            }
237                                        }
238                                    }
239                                    _ => {}
240                                }
241                            }
242                        }
243                        _ => {}
244                    }
245                }
246
247                if let Ok(msg) = thread_rx.try_recv() {
248                    match msg {
249                        ThreadMessage::RegisterHotKey(hotkey, tx) => {
250                            let _ = tx.send(register_hotkey(
251                                &xlib,
252                                display,
253                                root,
254                                &mut hotkeys,
255                                hotkey,
256                            ));
257                        }
258                        ThreadMessage::RegisterHotKeys(keys, tx) => {
259                            for hotkey in keys {
260                                if let Err(e) =
261                                    register_hotkey(&xlib, display, root, &mut hotkeys, hotkey)
262                                {
263                                    let _ = tx.send(Err(e));
264                                }
265                            }
266                            let _ = tx.send(Ok(()));
267                        }
268                        ThreadMessage::UnRegisterHotKey(hotkey, tx) => {
269                            let _ = tx.send(unregister_hotkey(
270                                &xlib,
271                                display,
272                                root,
273                                &mut hotkeys,
274                                hotkey,
275                            ));
276                        }
277                        ThreadMessage::UnRegisterHotKeys(keys, tx) => {
278                            for hotkey in keys {
279                                if let Err(e) =
280                                    unregister_hotkey(&xlib, display, root, &mut hotkeys, hotkey)
281                                {
282                                    let _ = tx.send(Err(e));
283                                }
284                            }
285                            let _ = tx.send(Ok(()));
286                        }
287                        ThreadMessage::DropThread => {
288                            (xlib.XCloseDisplay)(display);
289                            return;
290                        }
291                    }
292                }
293
294                std::thread::sleep(std::time::Duration::from_millis(50));
295            }
296        };
297    } else {
298        #[cfg(debug_assertions)]
299        eprintln!("Failed to open Xlib, maybe you are not running under X11? Other window systems on Linux are not supported by `global-hotkey` crate.");
300    }
301}

410fn modifiers_to_x11_mods(modifiers: Modifiers) -> u32 {
411    let mut x11mods = 0;
412    if modifiers.contains(Modifiers::SHIFT) {
413        x11mods |= xlib::ShiftMask;
414    }
415    if modifiers.intersects(Modifiers::SUPER | Modifiers::META) {
416        x11mods |= xlib::Mod4Mask;
417    }
418    if modifiers.contains(Modifiers::ALT) {
419        x11mods |= xlib::Mod1Mask;
420    }
421    if modifiers.contains(Modifiers::CONTROL) {
422        x11mods |= xlib::ControlMask;
423    }
424    x11mods
