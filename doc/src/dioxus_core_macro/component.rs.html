11impl Parse for ComponentBody {
12    fn parse(input: ParseStream) -> Result<Self> {
13        let item_fn: ItemFn = input.parse()?;
14        validate_component_fn(&item_fn)?;
15        Ok(Self { item_fn })
16    }

19impl ToTokens for ComponentBody {
20    fn to_tokens(&self, tokens: &mut TokenStream) {
21        // https://github.com/DioxusLabs/dioxus/issues/1938
22        // If there's only one input and the input is `props: Props`, we don't need to generate a props struct
23        // Just attach the non_snake_case attribute to the function
24        // eventually we'll dump this metadata into devtooling that lets us find all these components
25        //
26        // Components can also use the struct pattern to "inline" their props.
27        // Freya uses this a bunch (because it's clean),
28        // e.g. `fn Navbar(NavbarProps { title }: NavbarProps)` was previously being incorrectly parsed
29        if self.is_explicit_props_ident() || self.has_struct_parameter_pattern() {
30            let comp_fn = &self.item_fn;
31            tokens.append_all(allow_camel_case_for_fn_ident(comp_fn).into_token_stream());
32            return;
33        }
34
35        let comp_fn = self.comp_fn();
36
37        // If there's no props declared, we simply omit the props argument
38        // This is basically so you can annotate the App component with #[component] and still be compatible with the
39        // launch signatures that take fn() -> Element
40        let props_struct = match self.item_fn.sig.inputs.is_empty() {
41            // No props declared, so we don't need to generate a props struct
42            true => quote! {},
43
44            // Props declared, so we generate a props struct and then also attach the doc attributes to it
45            false => {
46                let doc = format!("Properties for the [`{}`] component.", &comp_fn.sig.ident);
47                let props_struct = self.props_struct();
48                quote! {
49                    #[doc = #doc]
50                    #[allow(missing_docs)]
51                    #props_struct
52                }
53            }
54        };
55
56        let completion_hints = self.completion_hints();
57
58        tokens.append_all(quote! {
59            #props_struct
60            #comp_fn
61
62            #completion_hints
63        });
64    }
65}

67impl ComponentBody {
68    // build a new item fn, transforming the original item fn
69    fn comp_fn(&self) -> ItemFn {
70        let ComponentBody { item_fn, .. } = self;
71        let ItemFn {
72            attrs,
73            vis,
74            sig,
75            block,
76        } = item_fn;
77        let Signature {
78            inputs,
79            ident: fn_ident,
80            generics,
81            output: fn_output,
82            ..
83        } = sig;
84
85        let Generics { where_clause, .. } = generics;
86        let (_, impl_generics, _) = generics.split_for_impl();
87        let generics_turbofish = impl_generics.as_turbofish();
88
89        // We generate a struct with the same name as the component but called `Props`
90        let struct_ident = Ident::new(&format!("{fn_ident}Props"), fn_ident.span());
91
92        // We pull in the field names from the original function signature, but need to strip off the mutability
93        let struct_field_names = inputs.iter().map(rebind_mutability);
94
95        let props_docs = self.props_docs(inputs.iter().collect());
96
97        let inlined_props_argument = if inputs.is_empty() {
98            quote! {}
99        } else {
100            quote! { #struct_ident { #(#struct_field_names),* }: #struct_ident #impl_generics }
101        };
102
103        // The extra nest is for the snake case warning to kick back in
104        parse_quote! {
105            #(#attrs)*
106            #(#props_docs)*
107            #[allow(non_snake_case)]
108            #vis fn #fn_ident #generics (#inlined_props_argument) #fn_output #where_clause {
109                {
110                    // In debug mode we can detect if the user is calling the component like a function
111                    dioxus_core::internal::verify_component_called_as_component(#fn_ident #generics_turbofish);
112                    #block
113                }
114            }
115        }
116    }
117
118    /// Build an associated struct for the props of the component
119    ///
120    /// This will expand to the typed-builder implementation that we have vendored in this crate.
121    /// TODO: don't vendor typed-builder and instead transform the tokens we give it before expansion.
122    /// TODO: cache these tokens since this codegen is rather expensive (lots of tokens)
123    ///
124    /// We try our best to transfer over any declared doc attributes from the original function signature onto the
125    /// props struct fields.
126    fn props_struct(&self) -> ItemStruct {
127        let ItemFn { vis, sig, .. } = &self.item_fn;
128        let Signature {
129            inputs,
130            ident,
131            generics,
132            ..
133        } = sig;
134
135        let struct_fields = inputs.iter().map(move |f| make_prop_struct_field(f, vis));
136        let struct_ident = Ident::new(&format!("{ident}Props"), ident.span());
137
138        parse_quote! {
139            #[derive(Props, Clone, PartialEq)]
140            #[allow(non_camel_case_types)]
141            #vis struct #struct_ident #generics {
142                #(#struct_fields),*
143            }
144        }
145    }
146
147    /// Convert a list of function arguments into a list of doc attributes for the props struct
148    ///
149    /// This lets us generate set of attributes that we can apply to the props struct to give it a nice docstring.
150    fn props_docs(&self, inputs: Vec<&FnArg>) -> Vec<Attribute> {
151        let fn_ident = &self.item_fn.sig.ident;
152
153        if inputs.is_empty() {
154            return Vec::new();
155        }
156
157        let arg_docs = inputs
158            .iter()
159            .filter_map(|f| build_doc_fields(f))
160            .collect::<Vec<_>>();
161
162        let mut props_docs = Vec::with_capacity(5);
163        let props_def_link = fn_ident.to_string() + "Props";
164        let header =
165            format!("# Props\n*For details, see the [props struct definition]({props_def_link}).*");
166
167        props_docs.push(parse_quote! {
168            #[doc = #header]
169        });
170
171        for arg in arg_docs {
172            let DocField {
173                arg_name,
174                arg_type,
175                deprecation,
176                input_arg_doc,
177            } = arg;
178
179            let arg_name = strip_pat_mutability(arg_name).to_token_stream().to_string();
180            let arg_type = crate::utils::format_type_string(arg_type);
181
182            let input_arg_doc = keep_up_to_n_consecutive_chars(input_arg_doc.trim(), 2, '\n')
183                .replace("\n\n", "</p><p>");
184            let prop_def_link = format!("{props_def_link}::{arg_name}");
185            let mut arg_doc = format!("- [`{arg_name}`]({prop_def_link}) : `{arg_type}`");
186
187            if let Some(deprecation) = deprecation {
188                arg_doc.push_str("<p>👎 Deprecated");
189
190                if let Some(since) = deprecation.since {
191                    arg_doc.push_str(&format!(" since {since}"));
192                }
193
194                if let Some(note) = deprecation.note {
195                    let note = keep_up_to_n_consecutive_chars(&note, 1, '\n').replace('\n', " ");
196                    let note = keep_up_to_n_consecutive_chars(&note, 1, '\t').replace('\t', " ");
197
198                    arg_doc.push_str(&format!(": {note}"));
199                }
200
201                arg_doc.push_str("</p>");
202
203                if !input_arg_doc.is_empty() {
204                    arg_doc.push_str("<hr/>");
205                }
206            }
207
208            if !input_arg_doc.is_empty() {
209                arg_doc.push_str(&format!("<p>{input_arg_doc}</p>"));
210            }
211
212            props_docs.push(parse_quote! { #[doc = #arg_doc] });
213        }
214
215        props_docs
216    }
217
218    fn is_explicit_props_ident(&self) -> bool {
219        if let Some(FnArg::Typed(PatType { pat, .. })) = self.item_fn.sig.inputs.first() {
220            if let Pat::Ident(ident) = pat.as_ref() {
221                return ident.ident == "props";
222            }
223        }
224
225        false
226    }
227
228    fn has_struct_parameter_pattern(&self) -> bool {
229        if let Some(FnArg::Typed(PatType { pat, .. })) = self.item_fn.sig.inputs.first() {
230            if matches!(pat.as_ref(), Pat::Struct(_)) {
231                return true;
232            }
233        }
234
235        false
236    }
237
238    // We generate an extra enum to help us autocomplete the braces after the component.
239    // This is a bit of a hack, but it's the only way to get the braces to autocomplete.
240    fn completion_hints(&self) -> TokenStream {
241        let comp_fn = &self.item_fn.sig.ident;
242        let completions_mod = Ident::new(&format!("{}_completions", comp_fn), comp_fn.span());
243
244        let vis = &self.item_fn.vis;
245
246        quote! {
247            #[allow(non_snake_case)]
248            #[doc(hidden)]
249            mod #completions_mod {
250                #[doc(hidden)]
251                #[allow(non_camel_case_types)]
252                /// This enum is generated to help autocomplete the braces after the component. It does nothing
253                pub enum Component {
254                    #comp_fn {}
255                }
256            }
257
258            #[allow(unused)]
259            #vis use #completions_mod::Component::#comp_fn;
260        }
261    }
262}

264struct DocField<'a> {
265    arg_name: &'a Pat,
266    arg_type: &'a Type,
267    deprecation: Option<crate::utils::DeprecatedAttribute>,
268    input_arg_doc: String,
269}
270
271fn build_doc_fields(f: &FnArg) -> Option<DocField> {
272    let FnArg::Typed(pt) = f else { unreachable!() };
273
274    let arg_doc = pt
275        .attrs
276        .iter()
277        .filter_map(|attr| {
278            // TODO: Error reporting
279            // Check if the path of the attribute is "doc"
280            if !is_attr_doc(attr) {
281                return None;
282            };
283
284            let Meta::NameValue(meta_name_value) = &attr.meta else {
285                return None;
286            };
287
288            let Expr::Lit(doc_lit) = &meta_name_value.value else {
289                return None;
290            };
291
292            let Lit::Str(doc_lit_str) = &doc_lit.lit else {
293                return None;
294            };
295
296            Some(doc_lit_str.value())
297        })
298        .fold(String::new(), |mut doc, next_doc_line| {
299            doc.push('\n');
300            doc.push_str(&next_doc_line);
301            doc
302        });
303
304    Some(DocField {
305        arg_name: &pt.pat,
306        arg_type: &pt.ty,
307        deprecation: pt.attrs.iter().find_map(|attr| {
308            if !attr.path().is_ident("deprecated") {
309                return None;
310            }
311
312            let res = crate::utils::DeprecatedAttribute::from_meta(&attr.meta);
313
314            match res {
315                Err(e) => panic!("{}", e.to_string()),
316                Ok(v) => Some(v),
317            }
318        }),
319        input_arg_doc: arg_doc,
320    })
321}

323fn validate_component_fn(item_fn: &ItemFn) -> Result<()> {
324    // Do some validation....
325    // 1. Ensure the component returns *something*
326    if item_fn.sig.output == ReturnType::Default {
327        return Err(Error::new(
328            item_fn.sig.output.span(),
329            "Must return a <dioxus_core::Element>".to_string(),
330        ));
331    }
332
333    // 2. make sure there's no lifetimes on the component - we don't know how to handle those
334    if item_fn.sig.generics.lifetimes().count() > 0 {
335        return Err(Error::new(
336            item_fn.sig.generics.span(),
337            "Lifetimes are not supported in components".to_string(),
338        ));
339    }
340
341    // 3. we can't handle async components
342    if item_fn.sig.asyncness.is_some() {
343        return Err(Error::new(
344            item_fn.sig.asyncness.span(),
345            "Async components are not supported".to_string(),
346        ));
347    }
348
349    // 4. we can't handle const components
350    if item_fn.sig.constness.is_some() {
351        return Err(Error::new(
352            item_fn.sig.constness.span(),
353            "Const components are not supported".to_string(),
354        ));
355    }
356
357    // 5. no receiver parameters
358    if item_fn
359        .sig
360        .inputs
361        .iter()
362        .any(|f| matches!(f, FnArg::Receiver(_)))
363    {
364        return Err(Error::new(
365            item_fn.sig.inputs.span(),
366            "Receiver parameters are not supported".to_string(),
367        ));
368    }
369
370    Ok(())
371}

373/// Convert a function arg with a given visibility (provided by the function) and then generate a field for the
374/// associated props struct.
375fn make_prop_struct_field(f: &FnArg, vis: &Visibility) -> TokenStream {
376    // There's no receivers (&self) allowed in the component body
377    let FnArg::Typed(pt) = f else { unreachable!() };
378
379    let arg_pat = match pt.pat.as_ref() {
380        // rip off mutability
381        // todo: we actually don't want any of the extra bits of the field pattern
382        Pat::Ident(f) => {
383            let mut f = f.clone();
384            f.mutability = None;
385            quote! { #f }
386        }
387        a => quote! { #a },
388    };
389
390    let PatType {
391        attrs,
392        ty,
393        colon_token,
394        ..
395    } = pt;
396
397    quote! {
398        #(#attrs)*
399        #vis #arg_pat #colon_token #ty
400    }
401}

403fn rebind_mutability(f: &FnArg) -> TokenStream {
404    // There's no receivers (&self) allowed in the component body
405    let FnArg::Typed(pt) = f else { unreachable!() };
406
407    let immutable = strip_pat_mutability(&pt.pat);
408
409    quote!(mut #immutable)
410}
411
412fn strip_pat_mutability(pat: &Pat) -> Pat {
413    let mut pat = pat.clone();
414    // rip off mutability, but still write it out eventually
415    if let Pat::Ident(ref mut pat_ident) = &mut pat {
416        pat_ident.mutability = None;
417    }
418
419    pat
420}

427fn keep_up_to_n_consecutive_chars(
428    input: &str,
429    n_of_consecutive_chars_allowed: usize,
430    target_char: char,
431) -> String {
432    let mut output = String::new();
433    let mut prev_char: Option<char> = None;
434    let mut consecutive_count = 0;
435
436    for c in input.chars() {
437        match prev_char {
438            Some(prev) if c == target_char && prev == target_char => {
439                if consecutive_count < n_of_consecutive_chars_allowed {
440                    output.push(c);
441                    consecutive_count += 1;
442                }
443            }
444            _ => {
445                output.push(c);
446                prev_char = Some(c);
447                consecutive_count = 1;
448            }
449        }
450    }
451
452    output
453}

455/// Takes a function and returns a clone of it where an `UpperCamelCase` identifier is allowed by the compiler.
456fn allow_camel_case_for_fn_ident(item_fn: &ItemFn) -> ItemFn {
457    let mut clone = item_fn.clone();
458    let block = &item_fn.block;
459
460    clone.attrs.push(parse_quote! { #[allow(non_snake_case)] });
461
462    clone.block = parse_quote! {
463        {
464            #block
465        }
466    };
467
468    clone
